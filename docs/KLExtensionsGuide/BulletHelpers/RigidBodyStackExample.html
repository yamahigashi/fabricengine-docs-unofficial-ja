/*

require Math;
require Geometry;
require BulletHelpers;


//
// Rb stack
//
function Init_RigidBodyStack(io RigidBodySimulation sim, Integer count)
{
  sim.initPhysics();

  sim.createGround();

  Scalar        mass = 10;

  Vec3 compoundCylinderDim = Vec3(4,1,1);
  Vec3 compoundBoxDim = Vec3(1,4,1);
  Vec3 boxDim = Vec3(1,1,1);
  Scalar sphereRadius = 1.5;

  BulletCompoundShape compoundShape =  BulletCompoundShape();
  BulletCollisionShape cylinderShape = BulletCylinderShapeX(compoundCylinderDim);
  BulletCollisionShape boxShape =  BulletBoxShape(compoundBoxDim);

  compoundShape.addChildShape(Xfo(), cylinderShape);
  compoundShape.addChildShape(Xfo(), boxShape);

  BulletCollisionShape  shapes[];
  shapes.push(compoundShape);
  shapes.push(BulletBoxShape(boxDim));
  shapes.push(BulletSphereShape(sphereRadius));

  PolygonMesh compoundMesh();
  compoundMesh.addCylinder(Xfo(Quat(Euler(Vec3(0, 0, HALF_PI)))), compoundCylinderDim.y, compoundCylinderDim.x * 2.0, true, 8, 2, true, false);
  compoundMesh.addCuboid(Xfo(), compoundBoxDim.x * 2.0, compoundBoxDim.y * 2.0, compoundBoxDim.z * 2.0, true, false);
  PolygonMesh cuboidMesh(); 
  cuboidMesh.addCuboid(Xfo(), boxDim.x * 2.0, boxDim.y * 2.0, boxDim.z * 2.0);
  PolygonMesh sphereMesh(); 
  sphereMesh.addSphere(Xfo(), sphereRadius, 12);

  PolygonMesh meshes[];
  meshes.push(compoundMesh);
  meshes.push(cuboidMesh);
  meshes.push(sphereMesh);

  for(Integer i=0;i<count;++i)
  {
    sim.addRigidBody(mass, 0.5, 0.0, Xfo(Vec3(0,6+6*i,0)), Vec3(), Vec3(), shapes[i%shapes.size()]);
  }
}


operator entry(){

  RigidBodySimulation sim();
  Init_RigidBodyStack(sim, 30);

  String reportedTransforms[];
  reportedTransforms.resize(sim.rigidBodies.size());

  for(Integer i=0; i<300; i++){
    sim.stepSimulation();

    if((i%10)==0){
      // Report X vertex positions;
      for(Integer k=0; k<sim.rigidBodies.size(); k++){
        BulletRigidBody rbm = sim.rigidBodies[k];
        reportedTransforms[k] = unitTestPrint(rbm.getWorldTransform());
      }
      report("i:"+i + " xfos:" + reportedTransforms);
    }
  }

  // Force the destruction of the sim object.
  sim = null;

  report("entry<<<<");
}
